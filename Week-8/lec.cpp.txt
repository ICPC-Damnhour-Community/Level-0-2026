void practice1(int n) {
    int count = 0;
    for (int i = 1; i <= n; i = i * 2) {
        count++;
    }
    cout << count << endl;
}

void practice2(int n) {
    for (int i = 0; i < n; i++) {
        cout << i << " ";
    }
    cout << endl;
    for (int j = 0; j < n; j++) {
        cout << j << " ";
    }

} //o(n)

void practice3(int n, int arr[]) {
    for (int i = 0; i < n; i++) {
        if (count(arr, arr + n, i) > 0) { //o(n)
            cout << "Found " << i << endl;
        }
    }
}  //o(n^2)

void practice4(int n, int m) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cout << i * j << " ";
        }
        cout << endl;
    }
}//o(n*m)
